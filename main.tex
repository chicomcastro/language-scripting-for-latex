% This is our scripting to set which language to use
% It's important on compilation only
% On interfaced editors, set this before compile
% On terminal, just compile with this commands:
% xelatex -jobname=<document-name> "<scripting line below> \input{document}"
% Where referenced script line is:
\newif\ifes\newif\ifen\newif\ifbr\brtrue
% Change its last statements for what language you want
% Ex.: \newif\ifes\newif\ifen\newif\ifbr\entrue (for english)

% We need to create these commands to use in text and automatically set
\newcommand{\langes}[1]{%
  \ifes\selectlanguage{spanish}#1\fi}
\newcommand{\langen}[1]{%
  \ifen\selectlanguage{english}#1\fi}
\newcommand{\langbr}[1]{%
  \ifbr\selectlanguage{brazilian}#1\fi}
% end-region <<Script for our language commands>>


% region <<Document>>
\begin{document}

\section{%
\langes{Hola!}
\langen{Hello!}
\langbr{Oi!}
}

\langes{Éste texto está en Español.\\
las reglas tipográficas españolas
\textit{deberían} aplicarse.}

\langen{This is text in English.\\
English typographic rules \textit{should} apply.}

\langbr{Esse texto está em Português.\\
As regras tipográficas brasileiras \textit{devem} aplicar-se.}

\end{document}
